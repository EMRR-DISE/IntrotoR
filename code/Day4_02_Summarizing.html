3 Grouping

# Group by Station
df_wq %>% group_by(Station) %>% summarize(mean_Chla = mean(Chla))
# A tibble: 2 × 2
  Station mean_Chla
  <chr>       <dbl>
1 D7           2.71
2 P8           3.37
# Calculate more than one summary statistic within `summarize`
df_wq %>% 
  group_by(Station) %>% 
  summarize(
    min_Chla = min(Chla),
    mean_Chla = mean(Chla),
    median_Chla = median(Chla),
    max_Chla = max(Chla),
    sd_Chla = sd(Chla)
  )
# A tibble: 2 × 6

df_wq_c <- df_wq %>% mutate(Year = year(Date))
